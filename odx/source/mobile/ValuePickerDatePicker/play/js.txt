require([
    "dojo/_base/window",
    "dojo/ready",
    "dojox/mobile/ValuePickerDatePicker",
    "dojox/mobile/parser"
], function(win, ready, ValuePickerDatePicker){
    ready(function(){
      var widget = new ValuePickerDatePicker({${props}}, "${id}");
      win.body().appendChild(widget.domNode);
      widget.startup();
    });
});
