require([
    "dojo/_base/window",
    "dojo/ready",
    "dojox/mobile/DatePicker",
    "dojox/mobile/parser"
], function(win, ready, registry, DatePicker){
    ready(function(){
      var widget = new DatePicker(${props}, "${id}");
      win.body().appendChild(widget.domNode);
      widget.startup();
    });
});
