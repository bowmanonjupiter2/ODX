require([
    "dojo/store/Memory", "dijit/form/FilteringSelect", "dojo/domReady!"
], function(Memory, FilteringSelect){

        data: [
            {name:"Apple", id:"ap"},
            {name:"Orange", id:"or"},
            {name:"Banana", id:"ba"},
            {name:"Strawberry", id:"st"},
            {name:"Pear", id:"pe"}

    });

    var filteringSelect = new FilteringSelect({${props},{store:fruitStore}}, "${id}").startup();
});
