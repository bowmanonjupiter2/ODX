require([
    "dijit/layout/BorderContainer", "dijit/layout/ContentPane",
    "dojo/domReady!"
], function(BorderContainer, ContentPane){
    var bc = new BorderContainer({
        ${props}, "${id}"
    });

    var cp1 = new ContentPane({
        region: "left",
        style: "width: 100px",
        content: "hello world"
    });
    bc.addChild(cp1);

    var cp2 = new ContentPane({
        region: "center",
        content: "how are you?"
    });
    bc.addChild(cp2);

    bc.placeAt(document.body);
    bc.startup();
});
