require([
    "dijit/form/MultiSelect", dojo/store/Memory", "dojo/domReady!"
], function(MultiSelect,Memory){

    var fruitStore = new Memory({
        data: [
            {name:"Apple", id:"ap"},
            {name:"Orange", id:"or"},
            {name:"Banana", id:"ba"},
            {name:"Strawberry", id:"st"},
            {name:"Pear", id:"pe"}
        ]
    });
    }
    var myMultiSelect = new MultiSelect({${props}, {store:fruitStore}}, "${id}").startup();

});
